 	freopen("BGAMES.INP","r",stdin);

  	scanf("%d%d", &n, &k);
  	for (int i = 0; i < n; ++i) {
    	scanf("%d", &h[i]);
    	c[h[i]]++;
    	gtk += h[i] > k;
  	}
  	for (int i = 0; i < n; ++i) {
    	int ret = 0;
    	if (h[i] < k) {
      		printf("0 ");
      		continue;
    	}
    	if (h[i] == k) {
      		printf("%d ", gtk);
      		continue;
    	}
    	for (int j = 1; j * j <= h[i] - k; ++j) {
      		if ((h[i] - k) % j != 0) continue;
      		if (h[i] % j == k) ret += c[j];
      		if (j * j != h[i] - k && h[i] % ((h[i] - k) / j) == k)
        		ret += c[(h[i] - k) / j];
    	}
    	if (k == 0) --ret;
    	printf("%d ", ret);
  	}
  	printf("\n");
  	return 0;
}

cint maxn = 3e5 + 7, maxh = 1e6 + 7;
int n, k, h[maxn], cnt[maxh] = {0}, ans[maxh] = {0}, d[maxh] = {0};

void bangk(){
    if (cnt[k] != 0){
        int dem = 0;
        loop(i, 1, n){
            if (h[i] > k) ++dem;
        }
        ans[k] = dem;
    }
}

void honk(){
    int c, n1, n2;
    loop(i, 1, n){
        if (h[i] > k && !d[h[i]]){
            d[h[i]] = 1, c = h[i] - k;
            loopdk(j, 1, j * j <= c, j++){
                if (c % j == 0){
                    if(j > k) ans[h[i]] += cnt[j];
                    if (c / j > k && j > 1) ans[h[i]] += cnt[c / j];
                }
            }
            if (k == 0) ans[h[i]] += cnt[h[i]] - 1;
        }
    }
}
