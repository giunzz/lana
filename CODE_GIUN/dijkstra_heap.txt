

int dijsktra(int startnode) {
    priority_queue < ii, vector < ii >, greater < ii > >  pq;
    pq.push(ii(startnode, 0));

    while (pq.empty() == false) {
        int u = pq.top().fi;
        int dis_u = pq.top().se;
        pq.pop();
        for (int i = 0; i < g[u].size(); ++i) {
            int v = g[u][i].fi;
            int w = g[u][i].se;
            if (dis[v] > dis_u + w) {
                dis[v] = dis_u + w);
                pq.push(ii(v, dis[v]));
            }
        }
    }
}
